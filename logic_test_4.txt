package day4logicTest;

public class SagregatePositiveAndNegative {
	
//	Q.1) Segregate positive and negative integers in linear time 
//	Given an array of positive and negative integers, 
//	. The output should print all negative numbers, followed 
//	by all positive numbers. 
//	For example, 
//	Input: [19, -13, 15, -12, -18, -16, 1, 3] 
//	Output: [-13, -12, -18, -16, 15, 19, 1, 3 

	public static void main(String[] args) {
		
		int arr[] = {19, -13, 15, -12, -18, -16, 1, 3};
		
		int left= 0;
		int right= arr.length-1;
		
		int res[] = new int[arr.length];
		for(int num:arr) {
			if(num <0) {
				res[left++] = num;
			}
			else {
				res[right--]= num;
			}
		}
		
		for(int num: res) {
			System.out.print(num+" ");
		}
	

	}

}

import java.util.Scanner;

public class DisplayEvenAtBeginningAndOddAtEnd {

	public static void main(String[] args) {
		// TODO Auto-generated method stub
		
//		Q.1) Accept 10 number in an array. Display all even number at the beginning and all Odd at 
//	the end. Use only one loop
		
		Scanner scanner = new Scanner(System.in);
		
		int arr[] = new int[10];
		
		int left =0;
		int right = arr.length-1;
		
		int res[] = new int[arr.length];
		
		System.out.println("Enter 10 numbers: ");
		for(int i=0; i<arr.length; i++) {
			arr[i] = scanner.nextInt();
		}
		
		System.out.println("Initial array: ");
		for(int num : arr) {
			System.out.print(num+" ");
		}
		
		
		for(int i: arr) {
			if(i%2==0) {
				res[left++]= i;
			}
			else {
				res[right--] = i;
			}
		}
		
		System.out.println("Output array: ");
		for(int i:res) {
			System.out.print(i+" ");
		}
		
		

	}

}



package day4logicTest;

import java.util.Arrays;
import java.util.Scanner;

public class binarySearch {

	public static void main(String[] args) {
		// TODO Auto-generated method stub
		
//		Q.2) Accept 5 number in an array and sort it. Accept a number from user and check if it is 
//	there in an array or not use binary search. 
		
		int arr[] = { 1,4,23,63,64,88};
		
		Arrays.sort(arr);
		
		Scanner sc = new Scanner(System.in);
		
		boolean flag=false;
		System.out.println("Enter a number: ");
		
		int target = sc.nextInt();
		
		int left =0; 
		int right = arr.length-1;
		
		while(left<=right) 
		{
			int mid = (left+right) / 2;
			
			if(arr[mid]==target) 
			{
				System.out.println("Number is present in array");
				flag=true;
				break;
			}
			
			else if(target < arr[mid]) {
				right = mid-1;
			}
			else {
				left= mid+1;
			}
			
		}
		
		if(flag==false) {
			System.out.println("Number is not present in array");
		}

	}

}
